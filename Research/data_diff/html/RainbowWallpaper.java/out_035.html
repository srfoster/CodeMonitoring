      387     void drawFrame(){
      388       final SurfaceHolder holder=getSurfaceHolder();
      389       Canvas c=null;
      390       try {
      391         c=holder.lockCanvas();
      392         if (c != null) {
      393           mRainbowPaint.setShader(null);
      394           if (mGradientType.equals(SOLID)) {
      395             drawColor(c);
      396           }

      399           }
      400  else if (mGradientType.equals(RADIAL)) {
      401             drawRadialGradient(c);

      404             drawSweepGradient(c);

      406  else if (mGradientType.equals(STRIPED_LINEAR)) {
      407             drawStripedLinear(c);
      408           }
      409  else if (mGradientType.equals(STRIPED_RADIAL)) {
      410             drawStripedRadial(c);
      411           }

      413             drawStripedSweep(c);
      414           }
      415           if (mDisplayShape) {
      416             drawShape(c);
      417           }
      418         }
      419       }
      420   finally {
      421         if (c != null) holder.unlockCanvasAndPost(c);
      422       }
      423       mHandler.removeCallbacks(mDrawColor);
      424       if (mVisible && (mFramesPerSecond > 0)) {
      425         mHandler.postDelayed(mDrawColor,1000 / mFramesPerSecond);
      426       }
